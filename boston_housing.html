def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    scores = [0,0,0]
    X_Test = []
    y_Test = []
    X_Train = []
    y_Train = []
    K = len(X)/cv
    X_Test.append(X[0:K:1])
    y_Test.append(y[0:K:1])
    X_Train.append(X[K::1])
    y_Train.append(y[K::1])
    
    X_Test.append(X[K:2*K:1])
    y_Test.append(y[K:2*K:1])
    X_Train.append(X[0:K:1])
    X_Train[1].extend = X[2*K::1]
    y_Train.append(y[0:K:1])
    y_Train[1].extend = y[2*K::1]
    
    X_Test.append(X[2*K::1])
    y_Test.append(y[2*K::1])
    X_Train.append(X[0:2*K:1])
    y_Train.append(y[0:2*K:1])
    
    for i in range(0,cv):
        estimator.fit(X_Train[i],y_Train[i])   
        scores[i] = performance_metric(y_Test[i], estimator.predict(X_Test[i]))
    
    return scores
